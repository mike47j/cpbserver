			for (rnd = 0; rnd < rounds.length; rnd++) {
			if (datafrom[ROUND] == rounds[rnd][RNDNAME])
				break;
			}
			var tb1str = "Hits", tb2str = "Golds";
			if (worldarchery==1) {
				tb1str = "Xs or 10s";
				tb2str = "Xs";
			}
			else if (rounds[rnd][RNDSCOREING]==Worcester) {
				tb2str = "5s";
			}
			else if (rounds[rnd][RNDSCOREING]!=Imperial)
				tb2str = "Xs or 10s";

            var txt = "<h2>Move " + datafrom[TARGET] + " to " + datato[TARGET] + "</h2>";

            txt += "<tr><td>Archer:</td><td>" + datafrom[NAME] + "</td><td>" + datato[NAME] + "</td></tr>";

            txt += "<td width=\"150\"><input type=\"submit\" value=\"Cancel\" formaction=\"/setup\" /></td>";
            txt += "</table></form>";
