function newtarget()
{
	var i, j, target;
	target = document.getElementById("target").value.toUpperCase();
	for (i = 0; i < free.length; i++) {
	for (j = 1; j < free[i].length; j++) {
		if (target == free[i][j]) {
		if (free[i][0] != "NotSet")
			data[ROUND] = free[i][0];
		data[TARGET] = target;
		page();
		return;
		}
	}
	}
}

function checksend() {
	var objRegExp = /^(\d{1,2})\/(\d{1,2})\/(\d{1,2})$/;
	var dob = document.getElementById("dob").value;
	var gender = document.getElementById("xgender").value
	if (dob!="") {
		if(!objRegExp.test(dob)) {
			document.getElementById("dob").style.background = "pink";
			return;
		}
		if (gender.substr(0,1)=="J") {
			document.getElementById("xgender").style.background = "pink";
			document.getElementById("dob").style.background = "pink";
			return;
		}
		var d = dob.match(objRegExp);
		dob = ("00"+d[1]).substr(d[1].length,2) + "/" + ("00"+d[2]).substr(d[2].length,2) + "/" + ("00"+d[3]).substr(d[3].length,2);
		document.getElementById("gender").value = gender.substr(0,1) + dob;
	}
	else
		document.getElementById("gender").value = gender;
	document.getElementById("send").value = "data";
	document.getElementById("data").submit();
}
